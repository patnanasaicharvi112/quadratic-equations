This is Problem-1(iv) from Exercise 4.2
\begin{enumerate}
\item ${16}x^2-{8} x +{1} = 0$\\
\solution\\
\begin{align}
x=\frac{-b\pm\sqrt{b^2-4ac}}{2a}\\
x=\frac{8\pm\sqrt{8^2-4 \times 16 \times 1}}{2 \times 16}\\                                                        x=\frac{8\pm\sqrt{64-64}}{32}\\
x=\frac{8\pm\sqrt{0}}{32}\\
x=\frac{8}{32}\\
x=\frac{1}{4}
\end{align}
\end{enumerate}
